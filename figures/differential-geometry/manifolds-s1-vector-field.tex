% A vector field on S^1: an outward tangent-ish arrow at several points.
% Source: book/tex/diffgeo/manifolds.tex (§ Tangent spaces and bundles).
\documentclass[border=2mm]{standalone}
\usepackage{tikz}\usetikzlibrary{arrows.meta}\usepackage{amssymb}
\begin{document}
\begin{tikzpicture}
  \draw[blue,line width=1pt] (0,0) circle (1); \node[blue] at (100:1.2){$S^1$};
  \foreach \a in {0,50,140,210,300}{
    \fill (\a:1) circle (1.4pt);
    \draw[-{Latex}] (\a:1)--(\a:1.5);}
\end{tikzpicture}
\end{document}

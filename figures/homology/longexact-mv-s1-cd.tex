% S^1 as two arcs c (left) and d (right) between a and b.
% Source: book/tex/homology/long-exact.tex (§ Application: Mayer-Vietoris).
\documentclass[border=2mm]{standalone}
\usepackage{tikz}
\usetikzlibrary{arrows.meta}
\begin{document}
\begin{tikzpicture}[>={Latex[length=2mm]}, scale=1.8]
  \node at (45:1) [above right] {$S^1$};
  \draw[->] (90:1) arc (90:270:1);
  \draw[->] (90:1) arc (90:-90:1);
  \fill (90:1) circle (0.9pt) node[above] {$a$};
  \fill (-90:1) circle (0.9pt) node[below] {$b$};
  \node[left] at (180:1) {$c$};
  \node[right] at (0:1) {$d$};
\end{tikzpicture}
\end{document}

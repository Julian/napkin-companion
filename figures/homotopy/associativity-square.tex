  \node[scale=0.7] at (0.625,0.88) {$\gamma_2$};
  \node[scale=0.7] at (0.875,0.88) {$\gamma_3$};
  \node[above,scale=0.7] at (0.5,1.12) {$\gamma_1\ast(\gamma_2\ast\gamma_3)$};

% Quotient of D^1 = [-1,1] by identifying its endpoints yields S^1.
% Source: book/tex/topology/constructions.tex (§ Quotient topology).
\documentclass[border=2mm]{standalone}
\usepackage{tikz}
\begin{document}
\begin{tikzpicture}[scale=1]
  % D^1 with endpoints -1, 1 marked
  \draw[blue] (-4.5,0) -- (-2,0) node[midway,below,blue] {$D^1$};
  \fill[red] (-4.5,0) circle (2.4pt) node[above,red] {$-1$};
  \fill[red] (-2,0) circle (2.4pt) node[above,red] {$1$};
  % arrow
  \draw[->] (-1.4,0) -- (-0.4,0);
  % S^1 with the fused point
  \draw[blue] (1.5,0) circle (1);
  \fill[red] (1.5,1) circle (2.4pt) node[above,red] {$-1\sim 1$};
  \node[blue] at (2.9,0.7) {$S^1$};
\end{tikzpicture}
\end{document}

% A lattice basis (alpha_1 = 1 + 3i, alpha_2 = 2 + 7i) whose
% fundamental parallelogram has mesh 1 even though neither basis
% vector is close to the origin.
% Source: book/tex/alg-NT/classgrp.tex (§ Third proof) — asy figure.
\documentclass[border=2mm]{standalone}
\usepackage{tikz}
\begin{document}
\begin{tikzpicture}[scale=0.42]
  \filldraw[fill=orange!20, draw=orange]
    (0,0) -- (1,3) -- (3,10) -- (2,7) -- cycle;
  \fill[blue] (1,3) circle (3pt) node[right, black] {$\alpha_1$};
  \fill[blue] (2,7) circle (3pt) node[above left, black] {$\alpha_2$};
  \draw[->] (-11,0) -- (11,0);
  \draw[->] (0,-11) -- (0,11);
\end{tikzpicture}
\end{document}

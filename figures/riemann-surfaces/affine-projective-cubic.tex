% The cubic curve ^3-x-y^2=0$ (an elliptic curve).
% Source: book/tex/riemann-surface/affine-projective.tex (§ Elliptic curves).
\documentclass[border=2mm]{standalone}
\usepackage{tikz}
\usepackage{amssymb}
\usetikzlibrary{arrows.meta}
\begin{document}
\begin{tikzpicture}[>=Latex,scale=0.9]
\draw[->](-2,0)--(3.2,0)node[right]{$x$};\draw[->](0,-2.6)--(0,2.6)node[above]{$y$};
\draw[red,thick,domain=-1:0,samples=40]plot(\x,{sqrt(\x*\x*\x-\x)});
\draw[red,thick,domain=-1:0,samples=40]plot(\x,{-sqrt(\x*\x*\x-\x)});
\draw[red,thick,domain=1:3,samples=60]plot(\x,{sqrt(\x*\x*\x-\x)});
\draw[red,thick,domain=1:3,samples=60]plot(\x,{-sqrt(\x*\x*\x-\x)});
\end{tikzpicture}
\end{document}

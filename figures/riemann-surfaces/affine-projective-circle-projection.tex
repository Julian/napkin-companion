% Parametrizing the upper unit circle by vertical projection to the 569Xaxis.
% Source: book/tex/riemann-surface/affine-projective.tex (§ Rational parametrization).
\documentclass[border=2mm]{standalone}
\usepackage{tikz}
\usepackage{amssymb}
\usetikzlibrary{arrows.meta}
\begin{document}
\begin{tikzpicture}[>=Latex,scale=1.6]
\draw(0,0)circle(1);
\foreach \x in {-0.95,-0.85,...,0.56}{\pgfmathsetmacro\y{sqrt(1-\x*\x)}\draw[blue,->](\x,\y)--(\x,0);\fill[blue](\x,\y)circle(0.8pt);\fill[blue](\x,0)circle(0.8pt);}
\draw[blue](-0.95,0)--(0.55,0);
\end{tikzpicture}
\end{document}
